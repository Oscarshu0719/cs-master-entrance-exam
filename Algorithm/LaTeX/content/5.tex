\item \begin{theorem}{(363)} Maximal points：
    \begin{algorithm}[H]
        \begin{algorithmic}[1]
            \Function{MaximalPoints}{Point [] $points$}
                \State $s$ := $\emptyset$
                \State Sort $points$ by x-coordinate in ascend order.
                \State $max\_y$ := $-\infty$
                \For {$i$ := $n$ to $1$}
                    \If {$points[i].y > max\_y$}
                        \State Add $points[i]$ to $s$.
                        \State $max\_y$ := $points[i].y$
                    \EndIf
                \EndFor
                \State \Return $s$
            \EndFunction
        \end{algorithmic}
    \end{algorithm}
\end{theorem}
