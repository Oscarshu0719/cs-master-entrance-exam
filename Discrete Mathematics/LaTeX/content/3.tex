\item \begin{theorem}{(3.56)} \quad\quad \begin{itemize}
        \item 若$A$為一集合，且$|A| = m$，$A$上等價關係的個數，即相異分割數。\begin{equation}
            \sum_{i = 1}^{m} \text{S}(m, i)
        \end{equation}
        \item $m$相異物放入$n$相同箱\textbf{可}空箱的方法。\begin{equation}
            \sum_{i = 1}^{n} \text{S}(m, i)
        \end{equation}
    \end{itemize}
\end{theorem}
