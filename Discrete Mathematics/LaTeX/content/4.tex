\item \begin{theorem}{(4.8)} 
    \begin{equation}
        \begin{aligned}
            \binom{-n}{r} & = (-1)^r\binom{n + r - 1}{r} \\
            (1 + x)^{-n} & = \sum_{r = 0}^{\infty}(-1)^r\binom{n + r - 1}{r} x^r
        \end{aligned}
    \end{equation}
\end{theorem}
